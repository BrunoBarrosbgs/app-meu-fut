
'use client';

import { useEffect, useState, useCallback } from 'react';
import type { Player, MatchModality } from '@/lib/types';
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Avatar, AvatarFallback, AvatarImage } from '@/components/ui/avatar';
import { X, Shield, Swords } from 'lucide-react';
import { Skeleton } from '@/components/ui/skeleton';
import { getPlayerTactic } from '@/lib/mock-responses';

interface IndividualTacticCardProps {
  player: Player;
  formation: string;
  modality: MatchModality;
  onClose: () => void;
}

export function IndividualTacticCard({ player, formation, modality, onClose }: IndividualTacticCardProps) {
    const [strengths, setStrengths] = useState<string>('');
    const [weaknesses, setWeaknesses] = useState<string>('');
    const [loading, setLoading] = useState(true);

    const getAnalysis = useCallback(() => {
        if (!player || !formation) return;

        setLoading(true);
        setTimeout(() => {
            const offensiveTactic = getPlayerTactic(formation, 'offensive', player.position);
            const defensiveTactic = getPlayerTactic(formation, 'defensive', player.position);
            setStrengths(offensiveTactic);
            setWeaknesses(defensiveTactic);
            setLoading(false);
        }, 300); // Simulate network delay
    }, [player, formation]);

    useEffect(() => {
        getAnalysis();
    }, [getAnalysis]);

    return (
        <Card className="flex flex-col h-full animate-in fade-in-50">
            <CardHeader className="relative">
                <Button variant="ghost" size="icon" className="absolute top-2 right-2 h-6 w-6" onClick={onClose}>
                    <X className="h-4 w-4" />
                </Button>
                <div className="flex flex-col items-center text-center">
                    <Avatar className="w-24 h-24 border-4 border-primary shadow-lg mb-2">
                        <AvatarImage src={player.avatarUrl} alt={player.name} data-ai-hint="player photo" />
                        <AvatarFallback>{player.name.charAt(0)}</AvatarFallback>
                    </Avatar>
                    <CardTitle className="text-2xl font-headline text-glow">{player.name}</CardTitle>
                    <CardDescription>{player.position} na formação {formation}</CardDescription>
                </div>
            </CardHeader>
            <CardContent className="flex-grow space-y-4">
                <h3 className="font-headline text-lg text-center text-glow">Instruções Táticas Individuais</h3>
                {loading ? (
                    <div className="space-y-4">
                        <div className="flex items-start gap-3">
                            <div className="p-1.5 bg-green-500/10 rounded-full border border-green-500/20 mt-1">
                                <Swords className="h-4 w-4 text-green-400" />
                            </div>
                            <div className="flex-grow space-y-2">
                                <Skeleton className="h-4 w-1/3" />
                                <Skeleton className="h-4 w-full" />
                            </div>
                        </div>
                         <div className="flex items-start gap-3">
                            <div className="p-1.5 bg-red-500/10 rounded-full border border-red-500/20 mt-1">
                                <Shield className="h-4 w-4 text-red-400" />
                            </div>
                            <div className="flex-grow space-y-2">
                                <Skeleton className="h-4 w-1/3" />
                                <Skeleton className="h-4 w-full" />
                            </div>
                        </div>
                    </div>
                ) : (
                    <div className="space-y-4 text-sm">
                        <div className="flex items-start gap-3">
                             <div className="p-1.5 bg-green-500/10 rounded-full border border-green-500/20 mt-1">
                                <Swords className="h-4 w-4 text-green-400" />
                            </div>
                            <div>
                                <h4 className="font-bold text-green-400">Instrução Ofensiva</h4>
                                <p className="text-muted-foreground">{strengths}</p>
                            </div>
                        </div>
                         <div className="flex items-start gap-3">
                            <div className="p-1.5 bg-red-500/10 rounded-full border border-red-500/20 mt-1">
                                <Shield className="h-4 w-4 text-red-400" />
                            </div>
                            <div>
                                <h4 className="font-bold text-red-400">Instrução Defensiva</h4>
                                <p className="text-muted-foreground">{weaknesses}</p>
                            </div>
                        </div>
                    </div>
                )}
            </CardContent>
        </Card>
    );
}
